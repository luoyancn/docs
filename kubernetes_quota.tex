\subsection{Quota}
\begin{code-block}{bash}
# 针对namespace设置quota
cat >namespace-quota.yml<<EOF
apiVersion: v1
kind: ResourceQuota
metadata:
  name: compute-resources
spec:
  hard:
    pods: "10"
    limits.cpu: "10"
    limits.memory: 10Gi
    limits.nvidia.com/gpu: 8

---

apiVersion: v1
kind: ResourceQuota
metadata:
  name: object-counts
spec:
  hard:
    configmaps: "10"
    persistentvolumeclaims: "10"
    replicationcontrollers: "10"
    secrets: "10"
    services: "10"
    services.loadbalancers: "0"
    services.nodeports: "10"
EOF
\end{code-block}
